18 /**
19  * A resettable monostable timer. Runs in conjunction with an endless loop
20  * which calls {@link #OnTimerTick()}.
21  *
22  * @author Peter Hoppe
23  */
24 public class TTickTimer
25 {
26  /**
27  * The delay time.
28  */
29  private long fExpireTime;
30 
31  /**
32  * The absolute time (since 1970-01-01_00:00 UTC) when the timer has been {@linkplain #Start() started}.
33  */
34  private long fStartTime;
35 
36  /**
37  * The state of the timer object.
38  */
39  private ETimerState fState;
40 
41  /**
42  * cTor.
43  *
44  * @param expireTime Time delay to expire after calling {@link #Start()}.
45  */
46  public TTickTimer (long expireTime)
47  {
48  fExpireTime = expireTime;
49  fStartTime = 0;
50  fState = ETimerState.kNull;
51  }
52 
53  /**
54  * @return <code>true</code> if this timer has expired.
55  */
56  public boolean HasExpired ()
57  {
58  boolean ret;
59 
60  ret = (fState == ETimerState.kExpired);
61 
62  return ret;
63  }
64 
65  /**
66  * The polling method. This timer is driven by calling this method in regular intervals.
67  */
68  public void OnTimerTick ()
69  {
70  long dT;
71  long t0;
72 
73  if (fState == ETimerState.kRunning)
74  {
75  t0 = System.currentTimeMillis ();
76  dT = t0 - fStartTime;
77  if (dT > fExpireTime)
78  {
79  fState = ETimerState.kExpired;
80  }
81  }
82  }
83 
84  /**
85  * Starts the timer.
86  */
87  public void Start ()
88  {
89  fStartTime = System.currentTimeMillis ();
90  fState = ETimerState.kRunning;
91  }
92 
93  /**
94  * Resets the timer back to null state, i.e. it's ready for the next {@link #Start()}.
95  */
96  public void Reset ()
97  {
98  fStartTime = 0;
99  fState = ETimerState.kNull;
100  }
101 }
