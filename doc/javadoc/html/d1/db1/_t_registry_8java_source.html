24 /**
25  * A registry for {@link VBrowseable browseable} Objects.
26  * Every object is registered by unique ID (Registry won't
27  * allow duplicate ID's). Clients can query and retrieve
28  * registered objects by their unique ID.
29  *
30  * @author Peter Hoppe
31  */
32 public class TRegistry
33 {
34  /**
35  * The audio driver.
36  */
37  private VAudioDriver fAudioDriver;
38 
39  /**
40  * Active frontend(s). These are kept in the {@link #fObjectsMap main}
41  * storage as well as in this storage. We need this extra bookkeeping
42  * so we can detect when the last frontend has been closed.
43  * In practice we'll use one front end only. But we still cater for
44  * the possibility to support multiple frontends.
45  */
46  private TArrayMap<VFrontend> fFrontends;
47 
48  /**
49  * The registry's main storage.
50  */
51  private TArrayMap<VBrowseable> fObjectsMap;
52 
53  /**
54  * cTor.
55  */
56  public TRegistry ()
57  {
58  fObjectsMap = new TArrayMap<> ();
59  fFrontends = new TArrayMap<> ();

95  /**
96  * Queries the central storage for object <code>id</code>.
97  *
98  * @param id ID of the object searched for.
99  * @return Queried object.
100  */
101  public VBrowseable GetObject (String id)
102  {
103  boolean hasObject;
104  VBrowseable ret;
105 
106  hasObject = fObjectsMap.HasElement (id);
107  if (hasObject)
108  {
109  ret = fObjectsMap.Get (id);
110  }
111  else
112  {
113  ret = null;
114  TLogger.LogWarning ("Failed to get object", this, "GetObject (" + id + ")");
115  }
116 
117  return ret;
118  }

120  /**
121  * Registers the audio driver. If another audio driver instance is
122  * already registered we'll log an error instead.
123  *
124  * @param d The driver to be registered.
125  */
126  public void Register (VAudioDriver d)
127  {
128  String id;
129 
130  if (fAudioDriver == null)
131  { /* No driver registered yet - go ahead. */
132  id = d.GetID ();
133  fAudioDriver = d;
134  fObjectsMap.Set (id, d);
135  TLogger.LogMessage ("Audio driver registered", this, "Register (" + id + ")");
136  }
137  else
138  { /* Another audio driver is already registered. */
139  TLogger.LogError
140  (
141  "Audio driver already assigned to module: '" + fAudioDriver.GetID () +
142  "'. Can't reassign to module '" + d.GetID () + "'",
143  this,
144  "Register (VBrowseable b)"
145  );
146  }
147  }
148

163  /**
164  * Registers the given frontend.
165  *
166  * @param fe Frontend to be registered.
167  */
168  public void Register (VFrontend fe)
169  {
170  String id;
171 
172  id = fe.GetID ();
173  fObjectsMap.Set (id, fe);
174  fFrontends.Set (id, fe);
175  TLogger.LogMessage ("Frontend registered", this, "Register (" + id + ")");
176  }
177 
178  /**
179  * Unregisters the object with the given ID.
180  *
181  * @param id ID of the object to be unregistered.
182  */
183  public void Unregister (String id)
184  {
185  boolean isFrontend;
186  boolean hasEntry;
187 
188  hasEntry = fObjectsMap.HasElement (id);
189  isFrontend = fFrontends.HasElement (id);
190  if (hasEntry)
191  {
192  fObjectsMap.Remove (id);
193  if (isFrontend)
194  {
195  fFrontends.Remove (id);
196  }
197 
198  if (isFrontend)
199  {
200  TLogger.LogMessage ("Frontend removed", this, "Unregister (" + id + ")");
201  }
202  else
203  {
204  TLogger.LogMessage ("Object removed", this, "Unregister (" + id + ")");
205  }
206  }
207  else
208  {
209  TLogger.LogWarning ("Failed (Unknown id)", this, "Unregister (" + id + ")");
210  }
211  }
