262  /**
263  * Auto compensation for Overruns/Underruns on the
264  * atomic buffer associated with the input.
265  */
266  private void _CompensateCongestions ()
267  {
268  VAudioPort_Input_Chunks_Buffered in;
269  TAtomicBuffer_Stats stats;
270  TAtomicBuffer_DumpRecord stRec;
271 
272  in = (VAudioPort_Input_Chunks_Buffered) GetPortIn (0);
273  stats = in.StatsGet ();
274  stRec = stats.GetRecord ();
275 
276  if (stRec.fDiffOverUnderruns > 0)
277  { /* Overruns - meaning the JackD audio driver pushes more
278  data than we handle at the moment => We must decrement
279  GUI timer cycle interval. We use a simple linear decrement. */
280  if (fTDeltaRequested > TTimer.gkLoopIntervalMin)
281  {
282  fTDeltaRequested--;
283  TController.PostEvent (gkEventTimerAdjustInterval, fTDeltaRequested, GetID ());
284  stats.Clear (); /* [150] */
285  }
286  }
287  else if (stRec.fDiffOverUnderruns < 0)
288  {
289  /* Underruns - meaning the GUI update cycle tries to draw too
290  * much data => We must increment GUI timer cycle interval.
291  * We use a simple linear increment. */
292  fTDeltaRequested++;
293  TController.PostEvent (gkEventTimerAdjustInterval, fTDeltaRequested, GetID ());
294  stats.Clear (); /* [150] */
295  }
296  }

298  /**
299  * Finds the raw peak value of (a part of) the given
300  * sample chunk and returns its absolute value.
301  * We only process up to <code>forNSamples</code>
302  * samples - in effect, not more samples than defined by
303  * {@link #fNSamplesPerCycle}.
304  *
305  * @param b The sample chunk.
306  * @param forNSamples How many samples we use to search.
307  * @return
308  */
309  private double _FindPeak (FloatBuffer b, long forNSamples)
310  {
311  boolean doContinue;
312  long nDone;
313  long nRem;
314  double x;
315  double ret;
316 
317  ret = 0;
318  nDone = 0;
319  nRem = b.remaining ();
320  doContinue = (nRem > 0);
321  while (doContinue)
322  {
323  x = b.get ();
324 
325  if (x < 0)
326  {
327  x = -x;
328  }
329 
330  if (x > ret)
331  {
332  ret = x;
333  }
334 
335  nDone++;
336  nRem = b.remaining ();
337  doContinue = ((nRem > 0) && (nDone < fNSamplesPerCycle));
338  }
339 
340  return ret;
341  }

343  /**
344  * The central computation piece of the PPM level meter. Sends one level update
345  * per cycle to the attached meter frontend. Output is one single floating
346  * point value per cycle. Level value is in dB. The method performs the
347  * necessary ballistics of a PPM meter.
348  *
349  * We only process up to {@link #fNSamplesPerCycle} samples during one cycle,
350  * whatever is left is processed during the next cycle(s). That way we can reconcile
351  * the given {@link #fSampleRate sample rate} and the calculated
352  * {@link #fNSamplesPerCycle samples per cycle}.
353  *
354  * New data always takes priority over old data. If new data comes in whilst
355  * there's still unprocessed data, we discard the old data and use the new data.
356  */
357  private void _ProcessChunk ()
358  {
359  TAudioContext_JackD drv;
360  VAudioPort_Input_Chunks_Buffered in;
361  TNodePPMProcessor_Endpoint_Out out;
362  FloatBuffer packetNew;
363  long nowT;
364  long dT;
365  double dY;
366  double dY_cycle;
367  double pdB;
368  float pSend;
369  long nSamples;
370  int nRem;
371 
372  drv = TController.GetAudioDriver ();
373  nowT = System.currentTimeMillis ();
374  dT = nowT - fTLast;
375  fTLast = nowT;
376  if (fTDeltaRequested <= -1)
377  { /* Will be -1 on first cycle only */
378  fTDeltaRequested = dT;
379  }
380  fSampleRate = drv.GetSampleRate ();
381  fNSamplesPerCycle = dT * fSampleRate / 1000;
382 
383  _CompensateCongestions ();
384 
385  fStats.SetCycleTime (dT);
386  fStats.SetSampleRate (fSampleRate);
387  fStats.SetNumSamplesPerCycle (fNSamplesPerCycle);
388 
389  /* Analyze next chunk of data. */
390  in = (VAudioPort_Input_Chunks_Buffered) GetPortIn (0);
391  packetNew = in.FetchPacket ();
392  if (packetNew != null)
393  {
394  /* We have fresh data. Do analysis with that one. */
395  fInPacket = packetNew;
396  fInPacket.rewind ();
397  nSamples = fInPacket.limit ();
398  if (nSamples > fNSamplesPerCycle)
399  {
400  nSamples = fNSamplesPerCycle;
401  }
402  fPeak_Raw = _FindPeak (fInPacket, nSamples);
403  }
404  else if (fInPacket != null)
405  {
406  /* No fresh data available, but the previous packet still has unresolved data */
407  fPeak_Raw = _FindPeak (fInPacket, fNSamplesPerCycle);
408  nRem = fInPacket.remaining ();
409  if (nRem <= 0)
410  {
411  fInPacket = null;
412  }
413  }
414  else
415  {
416  /* No data available for analysis. Just rebroadcast previous peak value. */
417  }
418 
419  /* Convert approximated peak value to dB */ /* [120] */
420  if (fPeak_Raw <= gkMinThreshold)
421  {
422  pdB = -130;
423  }
424  else
425  {
426  pdB = 20 * Math.log10 (fPeak_Raw);
427  }
428 
429  fStats.SetPeakValue (fPeak_Raw);
430  fStats.SetDBValue (pdB);
431 
432  /* Integration part - compute new peak value with given PPM ballistics. */
433  /* Interpolate next meter point. We use a simple linear interpolation. */
434  dY = pdB - fPeak_dB;
435  if (dY > 0)
436  {
437  /* Value is rising. Use value-rise ballistics. */
438  dY_cycle = dT * fPeakDeltaRise;
439  fPeak_dB = fPeak_dB + dY_cycle;
440 
441  /* Limit peak value in case it's above given peak parameter */
442  if (fPeak_dB > pdB)
443  {
444  fPeak_dB = pdB;
445  }
446  }
447  else if (dY < 0)
448  {
449  /* Value is falling. Use value-fall ballistics. */
450  dY_cycle = dT * fPeakDeltaFall;
451  fPeak_dB = fPeak_dB + dY_cycle;
452 
453  /* Limit peak value in case it's below given peak parameter */
454  if (fPeak_dB < pdB)
455  {
456  fPeak_dB = pdB;
457  }
458  }
459  else
460  {
461  /* Same value as before */
462  fPeak_dB = pdB;
463  }
464 
465  fStats.SetPeakValueProjected (fPeak_dB);
466 
467  /* Send peak value */
468  pSend = (float) fPeak_dB; /* [130] */
469  out = (TNodePPMProcessor_Endpoint_Out) GetPortOut (0);
470  out.PushSample (pSend);
471  }
472 }

474 /*
475 [110] PPM type II ballistics (strict):
476  Rise: -24dB -> - 1dB: 10 ms = 23dB / 10mS = 2.3 dB/mS
477  Fall: 0dB -> -24dB: 2800 ms = -24dB / 2800ms ~ -0.00857 dB/ms
478 
479 [120] Peak values, Raw values, Calibration and all that...
480 
481  For each sample, JJack delivers a floating point value
482  in the range [-1.0, 1.0] floating point units.
483  For each frame, we map all negative values to the
484  positive range, effectively rectifying the signal.
485  The positive maximum value in each frame becomes the
486  peak value.
487 
488  Peak value vs. Input signal voltage
489  -----------------------------------
490  Our calculations make the simplifying assumption
491  that the raw values delivered by jackd map directly
492  to voltage, e.g. 0.5 means: 0.5V. It's an arbitrary
493  assumption; in reality the voltage will differ wildly
494  from sound card to sound card. That's why we can't
495  use a unified calibration procedure. Future developments
496  could include an input gain setting that calibrates the
497  PPM to map peak value 1.0 against PPM 7 + 4dB (i.e.
498  an imaginary PPM 8) to add some headroom.
499  This is a very crude measurement. For now, if the sound
500  card input is connected to a mixer output, then use the
501  mixer's VU meter and match +4dB to PPM 7.
502 
503  Calculations
504  ------------
505  Range (peak value, fpU): floating point units, from JJack provided values
506  [0.0, 1.0]
507 
508  Generic formula: ratio (vdB): Signal Voltage (v) against reference voltage (vr):
509 
510  vdB = 20 * log (v / vr)
511 
512  Let vr = 1.0, then
513  vdB = 20 * log (v)
514 
515  Our mapping fpU -> dB:
516  fpU = [0.0, 10^-7.5]: -130
517  ]10^-7.5, 1]: -130 .. 0
518 
519  Note that
520  -130dB: Hearing threshold
521 
522  which computes to a voltage of:
523  10 ^ (-130/20) = 10^-7.5 = 3.16 * 10^-8
524 
525  Calculating peak values in dB:
526  for p: [0, 3.16E-08]
527  pdB = -130
528  for p: ]3.16E-08, 1.0]
529  pdB = 20 * log (p)
530 
531 [130] There are situations where downcasting from double to float
532  is dangerous. Here it's fairly safe:
533  * We deal with dB values, i.e. the really interesting part
534  is to the left of the decimal point. To the human ear,
535  -60dB isn't noticably different from -60.2dB.
536  * Round down errors (double -> float) happen in a dimension
537  somewhere below 1E-30 which makes no difference when
538  displayed on a meter display.
539  * We produce a use-once value, i.e. we won't use this value
540  as a base for the next peak calculations. Therefore we won't
541  have accumulating errors.
542 
543 [140] For time measurement. We will use the first cycle after starting
544  to just set an initial time point and otherwise do no data processing.
545  Otherwise, the first cycle could provide non-sensical values, maybe
546  even leading to an exception.
547 
548 [150] Otherwise stRec.fDiffOverUnderruns won't converge to zero without more over/underruns
549 */
