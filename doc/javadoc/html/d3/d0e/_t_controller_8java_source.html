726  /**
727  * Starts/stops all modules in the start list. Modules will be processed
728  * in the order the entries appear in the resp. list.
729  *
730  * @param doStart If <code>true</code>, start modules in the start list.
731  * Otherwise, stop modules in the stop list.
732  */
733  private void _StartStop (boolean doStart)
734  {
735  int i;
736  int n;
737  String verb;
738  String id;
739  ArrayList<String> list;
740  IControllable mod;
741 
742  verb = doStart ? "starting" : "stopping";
743  list = doStart ? fListIDModulesStart : fListIDModulesStop;
744  n = list.size ();
745  if (n >= 1)
746  {
747  for (i = 0; i < n; i++)
748  {
749  id = list.get (i);
750  TLogger.LogMessage (verb + ": module " + id);
751  mod = (IControllable) fRegistry.GetObject (id);
752  if (doStart)
753  {
754  mod.Start ();
755  }
756  else
757  {
758  mod.Stop ();
759  }
