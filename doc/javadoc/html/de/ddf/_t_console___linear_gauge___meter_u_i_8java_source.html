139  /**
140  * Creates a pre cached empty meter bar.
141  */
142  private void _PrecomputeFixedElements ()
143  {
144  int i;
145 
146  fEmptyBar = "";
147  for (i = 1; i <= fBarLen; i++)
148  {
149  fEmptyBar += " ";
150  }
151  }

153  /**
154  * Computes the level bar string for the given level value.
155  *
156  * @param level The level value.
157  */
158  private void _SetBar (float level)
159  {
160  double lDisp;
161  int i;
162  int nSegments;
163 
164  /* Compute progress bar value */
165  if (level > 0)
166  { /* ]0, ...] dB => hard limit to 100 */
167  lDisp = gkNumSections;
168  }
169  else if (level >= -24)
170  { /* [-24, 0] dB => working range 1..7 */ /* [100] */
171  lDisp = (level + 24) / 4 + 1;
172  }
173  else if (level >= -130)
174  {
175  /* [-130, -24[ dB => range 0..1 */ /* [110] */
176  lDisp = (level + 130) / 130;
177  }
178  else
179  {
180  /* below -130dB => hard limit to zero */
181  lDisp = 0;
182  }
183 
184  nSegments = (int) (fBarLen * lDisp/gkNumSections + 0.5);
185  fLevelBar = fEmptyBar + "\r" + fColorCode;
186  if (nSegments >= 1)
187  {
188  for (i = 1; i <= nSegments; i++)
189  {
190  fLevelBar += gkMeterBarChar;
191  }
192  fLevelBar += "\r" + fColorCodeReset;
193  }
194  }
195 
196  /**
197  * Sets the level warn and clip indication for the given level value.
198  * Clip will be cleared if the {@link #fClipTimer} has expired and the
199  * given level value is below clip/warn level.
200  *
201  * @param level The level value.
202  */
203  private void _SetClip (float level)
204  {
205  boolean hasExpired;
206 
207  fClipTimer.OnTimerTick ();
208  hasExpired = fClipTimer.HasExpired ();
209  if (level >= gkLvlClip)
210  {
211  fClipTimer.Start ();
212  fColorCode = gkAnsiRed;
213  fColorCodeReset = gkAnsiReset;
214  }
215  else if (level >= gkLvlWarn)
216  {
217  fClipTimer.Start ();
218  fColorCode = gkAnsiYellow;
219  fColorCodeReset = gkAnsiReset;
220  }
221  else if (hasExpired)
222  {
223  fClipTimer.Reset ();
224  fColorCode = "";
225  fColorCodeReset = "";
226  }
227  }
228 }
