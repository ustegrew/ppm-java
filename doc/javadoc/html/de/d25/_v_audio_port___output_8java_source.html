20 /**
21  * Base class for an output audio port.
22  * An output port is always associated with a hosting audio
23  * {@link VAudioProcessor processor}. It's connected with
24  * another {@link VAudioPort_Input} to where it sends its data.
25  * Every time this port sends data it will trigger a processing
26  * cycle in the audio processor associated with the connected
27  * input.<br/>
28  *
29  * To ensure type safety the connection mechanism uses
30  * the visitor pattern for the connection to an input port.
31  * This class is the visitor base class and provides the
32  * prototypes for the <code>_Visit(...)</code> methods.
33  *
34  * @author Peter Hoppe
35  */
36 public abstract class VAudioPort_Output extends VAudioPort
37 {
38  /**
39  * The input connected to this output.
40  */
41  private VAudioPort_Input fTarget;

94  {
95  String msg;
96 
97  msg = "Connecting endpoints: '"

101  + ") -> '"
102  + target.GetID ()
103  + "' ("
104  + target._GetType ()
105  + "): ";
106  if (isErr)
107  {
108  msg += "Failure: Incompatible endpoint types";
109  TLogger.LogWarning (msg, this, "SetTarget");
110  }
111  else
112  {
113  msg += "OK";
114  TLogger.LogMessage (msg, this, "SetTarget");
115  }
116  }
