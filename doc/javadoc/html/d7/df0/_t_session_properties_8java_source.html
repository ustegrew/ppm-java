187  /**
188  * Sets the options from the parsed commandline.
189  *
190  * @param cl Parsed commandline.
191  */
192  void Set (CommandLine cl)
193  {
194  boolean doContinue;
195  String v;
196 
197  fIsValid = true;
198  fErrorMsg = "";
199 
200  doContinue = true;
201 
202  /* Setting up: Option: -h | --help */
203  if (cl.hasOption ("h"))
204  {
205  fOptPrintHelp = true;
206  doContinue = false;
207  }
208 
209  /* Setting up: Option -l | --logFile */
210  if (doContinue)
211  {
212  if (cl.hasOption ("l"))
213  {
214  fOptLogFilePath = cl.getOptionValue ("l");
215  }
216  else
217  {
218  doContinue = false;
219  _Invalidate ("Missing option: 'l'");
220  }
221  }
222 
223  /* Setting up: Option: -u | --uiType */
224  if (doContinue)
225  {
226  if (cl.hasOption ("u"))
227  {
228  v = cl.getOptionValue ("u");
229  if (v.equals ("consoleLinear"))
230  {
231  fOptUIType = EFrontendType.kConsoleLinearGauge;
232  }
233  else if (v.equals ("consoleText"))
234  {
235  fOptUIType = EFrontendType.kConsoleTextOut;
236  }
237  else if (v.equals ("guiLinear"))
238  {
239  fOptUIType = EFrontendType.kGUILinearGauge;
240  }
241  else if (v.equals ("guiRadial"))
242  {
243  fOptUIType = EFrontendType.kGUINeedle;
244  }
245  else
246  {
247  System.err.println ("Error in option 'u'. Must be one of: 'consoleLinear', 'consoleText', 'guiLinear', 'guiRadial'. Set to 'consoleLinear'.");
248  }
249  }
250  else
251  {
252  System.err.println ("Missing option 'u'. Set to 'consoleLinear'.");
253  }
254  }
255 
256  /* Setting up: Option: -w | --consoleWidth */
257  if (doContinue)
258  {
259  if (cl.hasOption ("w"))
260  {
261  v = cl.getOptionValue ("u");
262  if (v.equals ("consoleLinear"))
263  {
264  if (cl.hasOption ("consoleWidth"))
265  {
266  v = cl.getOptionValue ("w");
267  try
268  {
269  fOptConsoleWidth = Integer.parseInt (v);
270  }
271  catch (NumberFormatException e)
272  {
273  System.err.println ("Invalid value for option 'w'. Set to default value (" + gkDefaultOptConsoleWidth + ")");
274  fOptConsoleWidth = gkDefaultOptConsoleWidth;
275  }
276  }
277  else
278  {
279  System.err.println ("Missing option 'w' (Should be specified if uiType = 'consoleLinear'). Set to default value (" + gkDefaultOptConsoleWidth + ")");
280  }
281  }
282  }
283  }
284 
285  /* Setting up: Option: -d | --debug */
286  if (doContinue)
287  {
288  if (cl.hasOption ("d"))
289  {
290  fOptShowDebugWindow = true;
291  }
292  }
293 
294  /* Setting up: Option: -x | --dPPMProc */
295  if (doContinue)
296  {
297  if (cl.hasOption ("x"))
298  {
299  fOptUseDeprecatedPPMProcessor = true;
300  }
301  }
302  }
