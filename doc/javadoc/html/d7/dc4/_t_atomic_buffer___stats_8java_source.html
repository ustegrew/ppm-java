29  /**
30  * Flag value for: Critical sections locked.
31  */
32  private static final int gkLocked = 1;
33 
34  /**
35  * Flag value for: Critical sections unlocked.
36  */
37  private static final int gkUnlocked = 0;
38 
39  /**
40  * The synchronization lock
41  */
42  private AtomicInteger fLock;
43 
44  /**
45  * The number of times the producer and the consumer clashed
46  * when trying to acquire the critical section. It's likely that there will
47  * be some contention, but a steep count rise means that producer/consumer
48  * are too aggressively trying to acquire the critical section.
49  * To reduce the rise of the contention count consider querying the lock
50  * state by having producer or consumer call {@link #IsLocked()}
51  * before trying to get/set sample data. You can also try to reduce the
52  * aggressiveness of both, the producer and the consumer - for example
53  * by having (one of) them cycle fewer times per second.
54  *
55  * @see TAtomicBuffer
56  */
57  private int fNumContentions;
58 
59  /**
60  * The number of overruns. Overruns happen if the producer
61  * sets more chunks than the consumer collects, resulting in dropped
62  * data (= data loss). See the {@link TAtomicBuffer}
63  * for an automated strategy to mitigate.
64  *
65  * @see TAtomicBuffer
66  */
67  private int fNumOverruns;
68 
69  /**
70  * The number of underruns. Underruns happen if the consumer
71  * tries to get more chunks than the producer submits, which leads
72  * to consumer starvation (= data loss). See the
73  * {@link TAtomicBuffer} for an automated strategy to
74  * mitigate.
75  *
76  * @see TAtomicBuffer
77  */
78  private int fNumUnderruns;
79 
80  /**
81  * The total number of contentions. This number will never be cleared.
82  *
83  * @see #fNumContentions
84  */
85  private long fTot_NumContentions;
86 
87  /**
88  * The total number of overruns. This number will never be cleared.
89  *
90  * @see #fNumOverruns
91  */
92  private long fTot_NumOverruns;
93 
94  /**
95  * The total number of underruns. This number will never be cleared.
96  *
97  * @see #fNumUnderruns
98  */
99  private long fTot_NumUnderruns;

124  * @see ppm_java.typelib.IStats#GetDumpStr()

126  @Override
127  public String GetDumpStr ()
128  {
129  int diff;
130  long totDiff;
131  String ret;
132 
133  _Lock ();
134 
135  diff = fNumOverruns - fNumUnderruns;
136  totDiff = fTot_NumOverruns - fTot_NumUnderruns;
137  ret = "cnt=" + fNumContentions + " (total:" + fTot_NumContentions + "), "
138  + "o_runs=" + fNumOverruns + " (total:" + fTot_NumOverruns + "), "
139  + "u_runs=" + fNumUnderruns + " (total:" + fTot_NumUnderruns + "), "
140  + "diff=" + diff + " (total:" + totDiff + ")";
141 
142  _Unlock ();
143

214  /**
215  * Increments the requested field.
216  *
217  * @param field Signifier of the field to be incremented.
218  */
219  private void _Increment (EIncField field)
220  {
221  _Lock ();
222 
223  if (field == EIncField.kContentions)
224  {
225  fNumContentions++;
226  fTot_NumContentions++;
227  }
228  else if (field == EIncField.kOverruns)
229  {
230  fNumOverruns++;
231  fTot_NumOverruns++;
232  }
233  else if (field == EIncField.kUnderruns)
234  {
235  fNumUnderruns++;
236  fTot_NumUnderruns++;
237  }
238 
239  _Unlock ();
240  }

242  /**
243  * Locks the critical sections. [100]
244  */
245  private void _Lock ()
246  {
247  boolean isLocked;
248 
249  isLocked = false;
250  while (! isLocked) /* [100] */
251  {
252  isLocked = fLock.compareAndSet (gkUnlocked, gkLocked);
253  }
254  }
255 
256  /**
257  * Unlocks the critical sections. [100]
258  */
259  private void _Unlock ()
260  {
261  boolean isUnlocked;
262 
263  isUnlocked = false;
264  while (! isUnlocked) /* [100] */
265  {
266  isUnlocked = fLock.compareAndSet (gkLocked, gkUnlocked);
267  }
268  }
269 }

271 /*
272 [100] For locking/unlocking: We use a simple spin lock as each critical section has
273  a linear execution flow and few statements, i.e. short execution time. Not
274  worth implementing anything more sophisticated.
275 */
