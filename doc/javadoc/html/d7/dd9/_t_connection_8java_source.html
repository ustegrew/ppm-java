22 /**
23  * A connection between two endpoints. This object just ensures that the
24  * connection is browseable. It won't handle any data flow.
25  *
26  * @author Peter Hoppe
27  */
28 public class TConnection extends VBrowseable
29 {
30  /**
31  * Creates a connection from endpoint <code>idFromPort</code> to
32  * endpoint <code>idToPort</code>.
33  *
34  * @param idFromPort Unique ID, source port.
35  * @param idToPort Unique ID, target port.
36  */
37  public static void CreateInstance (String idFromPort, String idToPort)
38  {
39  VAudioPort_Output fromPort;
40  VAudioPort_Input toPort;
41 
42  fromPort = (VAudioPort_Output) TController.GetObject (idFromPort);
43  toPort = (VAudioPort_Input) TController.GetObject (idToPort);
44  fromPort.SetTarget (toPort);
45  new TConnection (fromPort, toPort);
46  }
47 
48  /**

51  private static int gCounter = -1;

53  /**
