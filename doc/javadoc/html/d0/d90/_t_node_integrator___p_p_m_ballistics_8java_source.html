209  /**
210  * Updates input value, then recomputes output value to
211  * follow with the set rise/fall ballistics of a PPM. Will then
212  * push new output value to the output port.
213  *
214  * @param dBValue The new input value.
215  */
216  void SetRef (float dBValue)
217  {
218  double t0;
219  double dY;
220  double dT;
221  float y1;
222  TNodeIntegrator_PPMBallistics_Endpoint_Out out;
223 
224  t0 = System.currentTimeMillis ();
225  dY = dBValue - fVDB;
226  dT = t0 - fTLast;
227  fTLast = t0;
228 
229  if (dY > 0)
230  { /* Value has risen. Apply rise ballistics. */
231  fVDB = fVDB + fValueDeltaRise * dT;
232  if (fVDB > dBValue)
233  { /* Brickwall limiter */
234  fVDB = dBValue;
235  }
236  }
237  else if (dY < 0)
238  { /* Value has fallen. Apply fall ballistics. */
239  fVDB = fVDB + fValueDeltaFall * dT;
240  if (fVDB < dBValue)
241  { /* Brickwall limiter */
242  fVDB = dBValue;
243  }
244  }
245  /* No else clause [130] */
246 
247  fStats.SetTimeCycle (dT);
248  fStats.SetValueDBActual (fVDB);
249  fStats.SetValueDBRef (dBValue);
250 
251  y1 = (float) fVDB;
252  out = (TNodeIntegrator_PPMBallistics_Endpoint_Out) GetPortOut (0);
253  out.PushSample (y1);
254  }
255 }

258 /*
259 [110] PPM type II ballistics (strict):
260  Rise: -24dB -> - 1dB: 10 ms = 23dB / 10mS = 2.3 dB/mS
261  Fall: 0dB -> -24dB: 2800 ms = -24dB / 2800ms ~ -0.00857 dB/ms
262 
263 [120] Whilst developing the step integrator I learned that I must avoid
264  mixing variable classes in calculations. Early versions used
265  calculations such as this one:
266  fVDB = fVDB + dY * fValueDeltaRise
267  Where I mixed up the corrector internal variable (dY) with the output
268  variable (fVDB). It was a bad potshot to have something, but resulted in
269  erratic behaviour on the attached meter. By keeping dY out of the output
270  value calculation I have a smooth meter movement which follows the
271  PPM ballistics as I observed in the studio. It may be worthwhile to
272  find a better theoretical underpinning of the PPM ballistics by finding
273  the differential equations that drives the meter up rise and fall movement
274  and then to redesign this class as a one-step-per-cycle solver of the rise
275  and fall equations (e.g. Runge-Kutta method). It sound yummily interesting,
276  but for later.
277 
278 [130] Else clause - do nothing. If value doesn't change, just re-use what's there.
279 */
