22 /**
23  * A monostable timer for a clipping LED on our PPM GUI.
24  * Sets the clipping LED to the desired state for a certain
25  * time interval and resets it back to the clear state when
26  * the interval has passed.
27  *
28  * Internally, this timer is running in an endless loop and
29  * does the LED setting in response to incoming client
30  * requests. A termination request will terminate the loop (i.e.
31  * this timer. To coordinate all the LED setting we use a
32  * simple state machine which receives the requests and
33  * does the corresponding transistions.
34  *
35  * This is an older development, the {@link TConsole_LinearGauge}
36  * uses the next development - a {@link TTickTimer} - for the clip
37  * setting. That one doesn't use its own thread and therefore would
38  * be better to use here. Since this one works, and it's a prototype
39  * application we keep this solution.
40  *
41  * @author Peter Hoppe
42  */
43 class TGUINeedle_TimerClipping extends Thread
44 {

107  {

114  @Override
115  public void run ()
116  {
117  fTLast = System.currentTimeMillis ();
118  while (fState != gkStateTerm)
119  {
120  _DoTransition ();
121  try {Thread.sleep (gkIntervalLoop);} catch (InterruptedException e) {}
122  }
123 
124  }

126  /**
127  * Requests setting of the clipping LED to given level.
128  */
129  public void SetClip (EGUINeedle_Surrogate_ClipType cType)
130  {
131  if (cType == EGUINeedle_Surrogate_ClipType.kWarn)
132  {
133  fRequest.getAndSet (gkStateWarn);
134  }
135  else if (cType == EGUINeedle_Surrogate_ClipType.kError)
136  {
137  fRequest.getAndSet (gkStateErr);

139  }

141  /**

150  /**
151  * Automaton core method. Sets the various states in response to incoming
152  * client requests and sets the associated clipping LED accordingly.
153  */
154  private void _DoTransition ()
155  {
156  int request;
157 
158  request = fRequest.getAndAdd (0); /* [100] */
159  if (fState == gkStateClear)
160  {
161  if (request == gkStateClear)
162  {
163  }
164  else if (request == gkStateWarn)
165  {
166  fState = gkStateWarn;
167  _SetLED ();
168  }
169  else if (request == gkStateErr)
170  {
171  fState = gkStateErr;
172  _SetLED ();
173  }
174  else
175  {
176  fState = gkStateTerm;
177  }
178  }
179  else if (fState == gkStateWarn)
180  {
181  _TryReset ();
182  if (request == gkStateClear)
183  {
184  fState = gkStateClear;
185  _SetLED ();
186  }
187  else if (request == gkStateWarn)
188  {

192  {

194  _SetLED ();
195  }
196  else

200  }

240  /**

256  }

259  }

262  /**
263  * Reset the clip/warn condition if {@link #gkIntervalWait wait interval}
264  * has passed.
265  */
266  private void _TryReset ()
267  {
268  long t;
269  long dt;
270 
271  t = System.currentTimeMillis ();
272  dt = t - fTLast;
273  if (dt > gkIntervalWait)
274  {
275  fTLast = t;
276  fRequest.getAndSet (gkStateClear);
277  }
278  }
