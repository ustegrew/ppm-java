113  /**
114  * Receives a sample chunk, finds the peak value and pushes its
115  * absolute value to the output port.
116  *
117  * @param chunk The sample chunk analyzed.
118  */
119  void ReceivePacket (FloatBuffer chunk)
120  {
121  int i;
122  int nSamples;
123  float s;
124  TNodePeakEstimator_Endpoint_Out out;
125 
126  if (chunk != null)
127  {
128  nSamples = chunk.limit ();
129  if (nSamples >= 1)
130  {
131  fPeakLast = 0;
132  for (i = 0; i < nSamples; i++)
133  {
134  s = chunk.get (i);
135  if (s < 0)
136  {
137  s = -s;
138  }
139 
140  if (s > fPeakLast)
141  {
142  fPeakLast = s;
143  }
144  }
145  }
146  }
147 
148  fStats.SetPeak (fPeakLast);
149 
150  out = (TNodePeakEstimator_Endpoint_Out) GetPortOut (0); /* [100] */
151  out.PushSample (fPeakLast);
152  }
153 }
154 
155 
156 /*
157 [100] For each sample, JJack delivers a floating point value
158  in the range [-1.0, 1.0] floating point units.
159  For each frame, we map all negative values to the
160  positive range, effectively rectifying the signal.
161  The positive maximum value in each frame becomes the
162  peak value.
163 
164  Range (peak value, fpU): floating point units, from JJack provided values
165  [0.0, 1.0]
166 [110] We will push the last peak if no valid data came in. This is by design.
167 */
