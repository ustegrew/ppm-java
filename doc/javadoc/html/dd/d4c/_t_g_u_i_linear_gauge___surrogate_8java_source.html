150  /**
151  * For the given channel, sets the signal level.
152  *
153  * @param level The level, in dB.
154  * @param iChannel Zero based index of the channel.
155  */
156  void SetLevel (float level, int iChannel)
157  {
158  float lDisp;
159 
160  /* We update stats cycle time when channel #0 is being updated. */
161  if (iChannel == 0)
162  fStat.OnCycleTick ();
163 
164  /* Set clipping indicators. */
165  if (level >= gkLvlClip)
166  {
167  fGUI.ClippingSet (EGUILinearGauge_ClipType.kError, iChannel);
168  }
169  else if (level >= gkLvlWarn)
170  {
171  fGUI.ClippingSet (EGUILinearGauge_ClipType.kWarn, iChannel);
172  }
173 
174  /* Compute progress bar value */
175  if (level > 0)
176  { /* ]0, ...] dB => hard limit to maximum */
177  fStat.SetCalcSection (iChannel, 3);
178  lDisp = 7.0f;
179  }
180  else if (level >= -24)
181  { /* [-24, 0] dB => working range 1..7 */ /* [100] */
182  fStat.SetCalcSection (iChannel, 2);
183  lDisp = (level + 24) / 4 + 1;
184  }
185  else if (level >= -130)
186  {
187  /* [-130, -24[ dB => range 0..1 */ /* [110] */
188  fStat.SetCalcSection (iChannel, 1);
189  lDisp = (level + 130) / 130;
190  }
191  else
192  {
193  /* below -130dB => hard limit to zero */
194  fStat.SetCalcSection (iChannel, 0);
195  lDisp = 0;
196  }
197 
198  fStat.SetDBValue (iChannel, level);
199  fStat.SetDisplayValue (iChannel, lDisp);
200 
201  fGUI.SetLevel (lDisp, iChannel);
202  }
203 }

206 /*
207 [100] Each marker on a PPM scale means 4dB rise and maps to a step size of 100/7 on the GUI meter,
208  i.e. 100/7 stands for 4dB. Also, scale value 1 maps to -24 dB. Hence
209 
210 
211  x_gui = 100 + (lv_dB/4) * (100/7)
212 
213  e.g. for -24dB, x_gui = 100 + (-24/4) * (100/7) = 100 - 6*100/7 = 14.28
214 
215  Before the cast to integer we also add 0.5 to get a round figure.
216 
217 [110] Similar calculations, except dB values between -130dB and -24dB
218  map to PPM range 0..1, i.e. 0*100/7 ... 1*100/7
219 
220  We translate level by 130dB and cover a dynamic range from
221  130dB - 24dB = 106dB and compute our PPM value so that
222  -130dB => PPM(0), -24dB => PPM (1).
223  lDisp = (level + 130) / 106 => We optimize to
224  lDisp = level/106 + 130/106
225  = level/106 + 1.226
226 
227  Before the cast to integer we also add 0.5 to get a round figure.
228 */
