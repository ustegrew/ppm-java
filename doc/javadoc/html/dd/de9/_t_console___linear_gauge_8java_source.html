189  /**
190  * Receives a level update on the given channel.
191  *
192  * @param level The new PPM level.
193  * @param iChannel The channel. <code>0</code> for left, <code>1</code> for right.
194  */
195  void _Receive (float level, int iChannel)
196  {
197  String meterBar;
198 
199  if (iChannel == 0)
200  {
201  fMeterL.Receive (level);
202  meterBar = fMeterL.GetLevelBar ();
203  meterBar = gkAnsiCursorPosMeterL + meterBar + gkAnsiCursorPosBottom;
204  }
205  else
206  {
207  fMeterR.Receive (level);
208  meterBar = fMeterR.GetLevelBar ();
209  meterBar = gkAnsiCursorPosMeterR + meterBar + gkAnsiCursorPosBottom;
210  }
211 
212  _RefreshUI (); /* 110 */
213  System.out.print (meterBar);
214  }

216  /**
217  * initializes the UI - sanitized the bar length and prepares all
218  * pre cached decorations.
219  *
220  * @param barLen The bar length to be used. Will be sanitized:
221  * Bar length limited to given boundaries,
222  * {@link #gkBarLenMin} and {@link #gkBarLenMax},
223  * and divisible by {@link #gkNumSections}.
224  */
225  private void _Init (int barLen)
226  {
227  int i;
228  String spaces;
229 
230  /* Set bar length to sane value. Bar length must be divisable by gkNumSections */
231  if (barLen < gkBarLenMin)
232  {
233  fBarLen = gkBarLenMin;
234  }
235  else if (barLen > gkBarLenMax)
236  {
237  fBarLen = gkBarLenMax;
238  }
239  else
240  {
241  fBarLen = gkNumSections * (barLen / gkNumSections); /* [100] */
242  }
243  fNumColsPerSection = fBarLen / gkNumSections;
244 
245  /* Precompute fixed frame elements */
246  spaces = "";
247  for (i = 1; i <= fNumColsPerSection-1; i++)
248  {
249  spaces += " ";
250  }
251 
252  fFramebar = gkTickmarkChar;
253  for (i = 1; i <= gkNumSections; i++)
254  {
255  fFramebar += spaces + gkTickmarkChar;
256  }
257 
258  fNumBar = "0";
259  for (i = 1; i <= gkNumSections; i++)
260  {
261  fNumBar += spaces + i;
262  }
263 
264  fEmptyBar = "";
265  for (i = 1; i <= fBarLen; i++)
266  {
267  fEmptyBar += " ";
268  }
269  }

283  System.out.println (fNumBar);
284  }

286  /**
287  * If the refresh timer has expired: Redraws the
288  * UI and restarts the timer [110].
289  */
290  private void _RefreshUI ()
291  {
292  boolean hasExpired;
293 
294  fRefreshTimer.OnTimerTick ();
295  hasExpired = fRefreshTimer.HasExpired ();
296  if (hasExpired)
297  {
298  _RefreshFrame ();
299  fRefreshTimer.Start ();
300  }
301  }

304 /*
305 [100] Guarantees that the bar length is an exact multiple of gkNumSections
306 
307 [110] The associated JJack library gave problems with the console output
308  by writing output to the console and mess up the UI (Text written underneath
309  and location of meter bar offset). For this reason we redraw the
310  entire UI in a regular interval and place the meter elements at an
311  absolute position.
312 */
