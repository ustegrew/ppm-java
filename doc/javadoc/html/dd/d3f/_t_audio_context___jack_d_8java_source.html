180  /**
181  * Loads and initializes the JackD driver.
182  */
183  private void _LoadDriver ()
184  {
185  String idClient;
186  int nPortsIn;
187  int nPortsOut;
188  int nPorts;
189 
190  idClient = GetID ();
191  nPortsIn = GetNumPortsIn ();
192  nPortsOut = GetNumPortsOut ();
193  nPorts = nPortsIn + nPortsOut;
194 
195  /* Setting properties for the underlying JJack framework. */
196  System.setProperty ("jjack.client.name", idClient);
197  System.setProperty ("jjack.ports", Integer.toString (nPorts));
198  System.setProperty ("jjack.ports.in", Integer.toString (nPortsOut));
199  System.setProperty ("jjack.ports.out", Integer.toString (nPortsIn));
200  System.setProperty ("jjack.ports.autoconnect", Boolean.toString (false));
201  System.setProperty ("jjack.ports.input.autoconnect", Boolean.toString (false));
202  System.setProperty ("jjack.ports.output.autoconnect", Boolean.toString (false));
203 
204  TLogger.LogMessage ("Loading JackD audio bridge.", this, "_LoadDriver ()");
205  /* Dummy call to load JJack driver. JJackSystem has a static
206  * block at the beginning which loads the JNI system library */
207  JJackSystem.class.getName ();
208  fIsWorking = JJackSystem.isInitialized ();
209 
210  if (fIsWorking)
211  {
212  fSampleRate = JJackSystem.getSampleRate ();
213 
214  /* Plug this instance into JackD. From now on process(JJackAudioEvent)
215  * will be called by JackD, in regular intervals. */
216  TLogger.LogMessage ("Activating JackD driver.", this, "_LoadDriver");
217  TLogger.LogMessage (JJackSystem.getInfo ());
218 
219  JJackSystem.setProcessor (this);
220  }
221  else
222  {
223  TLogger.LogError ("JackD driver did not load successfully", this, "_LoadDriver");
224  }
225  }

227  /**
228  * Processes data ready to be sent to the Jack driver i.e. towards the JackD server).
229  * This sounds paradox, but we define ports as INPUTS which receive data from our
230  * program and send it to the JackD server.
231  *
232  * We use a very crude overflow protection - this saves us lots of code which has
233  * to deal with exceptions, resp. loop over the buffered samples etc. This is
234  * time critical data handling, therefore we want to keep it as simple as possible.
235  *
236  * The client program is responsible to ensure that data frames don't exceed the
237  * prescribed buffer length.
238  *
239  * @param e The handle allowing access to the input buffers.
240  */
241  private void _ProcessInputs (JJackAudioEvent e)
242  {
243  TAudioContext_Endpoint_In portIn;
244  int nPorts;
245  int i;
246  int nSampIn;
247  int nSampOut;
248  FloatBuffer bIn;
249  FloatBuffer bOut;
250 
251  nPorts = GetNumPortsIn ();
252  if (nPorts >= 1)
253  {
254  for (i = 0; i < nPorts; i++)
255  {
256  portIn = (TAudioContext_Endpoint_In) GetPortIn (i);
257  bIn = portIn.FetchPacket ();
258  bOut = e.getOutput (i);
259  nSampIn = bIn.capacity ();
260  nSampOut = bOut.capacity ();
261 
262  /* [100] */
263  if (nSampIn <= nSampOut)
264  {
265  bOut.put (bIn);
266  }
267  }
268  }
269  }
270 
271  /**
272  * Processes data ready to be retrieved from the Jack driver i.e. away from
273  * the JackD server and towards the client program.
274  * This sounds paradox, but we define ports as OUTPUTS which receive data
275  * from the JackD server and send it to the client program.
276  *
277  * @param e The handle allowing access to the output buffers.
278  */
279  private void _ProcessOutputs (JJackAudioEvent e)
280  {
281  TAudioContext_Endpoint_Out portOut;
282  int nPorts;
283  int i;
284  FloatBuffer b;
285 
286  nPorts = GetNumPortsOut ();
287  if (nPorts >= 1)
288  {
289  for (i = 0; i < nPorts; i++)
290  {
291  portOut = (TAudioContext_Endpoint_Out) GetPortOut (i);
292  b = e.getInput (i);
293  portOut.PushPacket (b);
294  }
295  }
296  }

316 /*
317 [100] We simply discard longer input buffers. This saves us having to
318  deal with a BufferOverflowException or having to execute code which
319  puts data up to a limit etc. This is time critical code, the less
320  it has to do the better!
321  The client is responsible to ensure that the number of frames is
322  correct.
323 
324 [110] To clear up misconception - this program (ppm) only needs an audio driver with outputs. Why
325  does the client have to specify inputs for the driver as well? Reason is: Because we
326  want this driver to be generic, i.e. capable to handle situations where we do need to
327  pass audio data back to the jackd server!
328 */
