128  public void SetLevel (double lvl, int iChannel)
129  {
130  LinearBargraph targetGUI;
131

142  }
143 
144  targetGUI.setValue (lvl);

156  /**
157  * Render clipping/warning level on LED of given channel.
158  *
159  * @param cType Clip type.
160  * @param iChannel Channel ID (<code>0</code>: Left channel, <code>1</code>: Right channel).
161  */
162  void _SetClipping (EGUILinearGauge_ClipType cType, int iChannel)
163  {
164  LinearBargraph meter;
165  LedColor color;
166  boolean isOn;
167 
168  switch (cType)
169  {
170  case kClear:
171  color = kColorNormal;
172  isOn = false;
173  break;
174  case kWarn:
175  color = kColorWarn;
176  isOn = true;
177  break;
178  case kError:
179  color = kColorClip;
180  isOn = true;
181  break;
182  default:
183  color = kColorNormal;
184  isOn = false;
185  }
186 
187  if (iChannel == gkIChanL)
188  {
189  meter = fMeterL;
190  }
191  else if (iChannel == gkIChanR)
192  {
193  meter = fMeterR;
194  }
195  else
196  {
197  meter = null;
198  }
199 
200  if (meter != null)
201  {
202  meter.setUserLedColor (color);
203  meter.setUserLedOn (isOn);
204  }
205  }
