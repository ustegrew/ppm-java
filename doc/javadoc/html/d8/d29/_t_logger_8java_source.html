164  /**
165  * Does any necessary setup work.
166  *
167  * @param target The target to send log messages to.
168  * @param filePath The log file to write to, if target is
169  * {@link ELogTarget#kFile}. Otherwise ignored.
170  */
171  private void _Init (ELogTarget target, String filePath)
172  {
173  Logger rootLogger;
174  Handler[] handlers;
175  int i;
176  int nHandlers;
177  Handler cH;
178  FileHandler fH;
179  SimpleFormatter sF;
180 
181  if (target == ELogTarget.kFile)
182  {
183  /* Remove console handlers; we write log entries to a log file */
184  rootLogger = Logger.getLogger ("");
185  handlers = rootLogger.getHandlers ();
186  nHandlers = handlers.length;
187  if (nHandlers >= 1)
188  {
189  for (i = 0; i < nHandlers; i++)
190  {
191  cH = handlers [i];
192  if (cH instanceof ConsoleHandler)
193  {
194  rootLogger.removeHandler (cH);
195  }
196  }
197  }
198 
199  try
200  {
201  sF = new SimpleFormatter ();
202  fH = new FileHandler (filePath);
203  fH.setFormatter (sF);
204  fLogger = Logger.getLogger (Logger.GLOBAL_LOGGER_NAME);
205  fLogger.addHandler (fH);
206  }
207  catch (SecurityException | IOException e)
208  {
209  e.printStackTrace();
210  }
211  }
212  else
213  {
214  fLogger = Logger.getGlobal ();
215  }
216  }

218  /**
219  * Writes a log entry. Depending on the log target, the message will either be written
220  * to a log file or to stdout/stderr.
221  *
222  * @param msg The message to log
223  * @param lv The log level
224  * @param source Name of the class from which the log entry originates.
225  * @param method Name of the method from which the log entry originates.
226  */
227  private void _Log (String msg, ELogLevel lv, Object source, String method)
228  {
229  String src;
230  String m;
231  String dtime;
232 
233  src = "";
234  if (source != null)
235  {
236  src = source.getClass ().getCanonicalName ();
237  }
238 
239  if (method != null)
240  {
241  src += "::" + method;
242  }
243 
244  dtime = new Date().toString ();
245  m = dtime + ": " + src + ": " + msg;
246  switch (lv)
247  {
248  case kError:
249  fLogger.severe (m);
250  break;
251  case kMessage:
252  fLogger.info (m);
253  break;
254  case kWarn:
255  fLogger.warning (m);
256  break;
257  default:
258  fLogger.info (m);
259  }
260  }
