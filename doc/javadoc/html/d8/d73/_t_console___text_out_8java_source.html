23 /**
24  * A simple text-only frontend. Writes a stream of records in JSON format
25  * to STDOUT. Caters for a special use case where we want to stream the metering
26  * information over a network connection to another host. As the JSON records
27  * are written to STDOUT it's possible to pipe them to another program,
28  * e.g. a web socket provider. This frontend produces two types of records:
29  * <dl>
30  * <dt>Sample record</dt>
31  * <dd>
32  * A record containing a value (sample). Contains the sample value
33  * and which channel it's for. Record scheme:
34  * <pre>
35  * {
36  * "type": "sample", // Record type ID
37  * "value": float-value, // The sample value
38  * "port": int-value // For which port (0: Left channel, 1: Right channel)
39  * }
40  * </pre>
41  * </dd>
42  * <dt>Info record</dt>
43  * <dd>
44  * A record containing in-band information (e.g. "Metering stops"). Record scheme:
45  * <pre>
46  * {
47  * "type": "info", // Record type ID
48  * "value": string // The info message
49  * }
50  * </pre>
51  * To create info records we use ppm-java's event system. If you use this frontend, make
52  * sure that you connect the frontend instance to an event emitter that sends the
53  * control messages.<br/>
54  * <b>Note that the info record API isn't fully developed yet.</b>
55  * </dd>
56  * </dl>
57  *
58  * @author Peter Hoppe
59  *
60  * TODO This frontend needs further development, as we haven't yet fully developed the info
61  * record API. So far, only "Start" and "Stop" are implemented. More items are needed,
62  * e.g. "is the audio in mono or stereo?"
63 
64  */

191  /**
192  * Creates a JSON record from the given parameters.
193  *
194  * @param fieldType Type of record.
195  * @param info If it's an info record, this is the info payload.
196  * @param sample If it's a sample record, this is the sample value.
197  * @param iPort If it's a sample record, the port the sample is for (L or R).
198  */
199  private void _PushRecord (ERecordType fieldType, String info, float sample, int iPort)
200  {
201  String record;
202 
203  if (fieldType == ERecordType.kInfo)
204  {
205  record =
206  "{" +
207  "'type':'" + gkRecordTypeInfo + "'," +
208  "'value':'" + info + "'" +
209  "}";
210  }
211  else if (fieldType == ERecordType.kSample)
212  {
213  record =
214  "{" +
215  "'type':'" + gkRecordTypeSample + "'," +
216  "'value':'" + sample + "'," +
217  "'port':'" + iPort + "'" +
218  "}";
219  }
220  else
221  {
222  record = "{'type':'null'}";
223  }
224 
225  System.out.println (record);
226  }
227 }
