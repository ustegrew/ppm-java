63  /* (non-Javadoc)
64  * @see ppm_java.typelib.IStats#GetDumpStr()
65  */
66  @Override
67  public String GetDumpStr ()
68  {
69  int i;
70  int n;
71  TGUINeedle_Surrogate_Stats_Record r;
72  String ret;
73 
74  ret = "GUI [" + fHost.GetID () + "]:\n" +
75  " cycleTime [ms] = " + fTimeCycle.getAndAdd (0) + "\n";
76 
77  n = fStatChannels.size ();
78  if (n >= 1)
79  {
80  ret += " Channels:\n";
81  for (i = 0; i < n; i++)
82  {
83  r = fStatChannels.get (i);
84  ret += " Channel [" + i + "]:\n" + r.GetDumpStr ();
85  }
86  }
87  else
88  {
89  ret += " Channels: None\n";
90  }
91 
92  return ret;
93  }
