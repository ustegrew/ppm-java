23 /**
24  * Internal worker thread. Runs in an endless loop and triggers
25  * the {@link IEvented#gkEventTimerTick} events. This worker can
26  * only run once, i.e. has no suspend state.
27  *
28  * @author Peter Hoppe
29  */
30 class TTimerWorker extends Thread
31 {
32  /**
33  * Minimum Interval.
34  */
35  private static final int gkLoopInterval = TTimer.gkLoopIntervalMin;
36 
37  /**
38  * Internal state: Worker thread is running.
39  */
40  private static final int gkStateRun = 1;
41 
42  /**
43  * Internal state: Worker thread is being stopped / is stopped.
44  */
45  private static final int gkStateStop = 2;
46 
47  /**
48  * Internal state: Worker thread is in wait state, i.e. not yet started.
49  */
50  private static final int gkStateWait = 0;
51 
52  /**
53  * The hosting timer module.
54  */
55  private TTimer fHost;
56 
57  /**
58  * The loop interval. Can be changed. To make it thread safe we choose an atomic variable.
59  */
60  private AtomicLong fIntervalMs; /* [100] */
61 
62  /**
63  * The internal state. Changed from outside. To make it thread safe we choose an atomic variable.
64  */
65  private AtomicInteger fState; /* [100] */
66 
67  /**
68  * cTor.
69  *
70  * @param host The hosting timer module.
71  * @param delayMs The initial interval.
72  */
73  TTimerWorker (TTimer host, long delayMs)
74  {
75  fHost = host;
76  fState = new AtomicInteger (gkStateWait);
77  fIntervalMs = new AtomicLong (delayMs);
78  }

80  /**
81  * @return The current interval.
82  */
83  public long GetInterval ()
84  {
85  long ret;
86 
87  ret = fIntervalMs.getAndAdd (0); /* [110] */
88 
89  return ret;
90  }

92  /* (non-Javadoc)
93  * @see java.lang.Runnable#run()
94  */
95  @Override
96  public void run ()
97  {
98  long t0;
99  long t1;
100  long dly;
101  long dt;
102  int state;
103 
104  fState.getAndSet (gkStateRun);
105  t0 = System.currentTimeMillis ();
106  do
107  {
108  try {Thread.sleep (gkLoopInterval);} catch (InterruptedException e) {}
109 
110  t1 = System.currentTimeMillis ();
111  dly = fIntervalMs.getAndAdd (0); /* [110] */
112  dt = t1 - t0;
113  if (dt >= dly)
114  {
115  fHost.SendTimerEvent ();
116  t0 = System.currentTimeMillis ();
117  }
118  state = fState.getAndAdd (0); /* [110] */
119  } while (state == gkStateRun);
120  }
121 
122  /**
123  * Sets ithe interval to a new value.
124  *
125  * @param intervalMs The new interval value.
126  */
127  void SetInterval (long intervalMs)
128  {
129  fIntervalMs.getAndSet (intervalMs);
130  }
131 
132  /**
133  * Stops this thread, i.e. terminates it. The worker thread can't be restarted.
134  */
135  void Stop ()
136  {
137  fState.compareAndSet (gkStateRun, gkStateStop); /* [120] */
138  }
139 }

141 /*
142 [100] We use atomic variables because the values can be accessed by multiple threads.
143 [110] A weird way to retrieve the value of fState - return value and add zero.
144  But the AtomicInteger::get() method seems to be non-atomic (according to the docs),
145  whilst AtomicInteger::getAndAdd() is atomic.
146 [120] compareAndSet: Because we don't set the state flag to gkStateStop unless the worker
147  is in gkStateRun, i.e. running. For now it seems unnecessary to return
148  whether setting was successful or not (compareAndSet returns true upon
149  success and false upon failure).
150 */
