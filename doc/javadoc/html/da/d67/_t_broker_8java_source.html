22 /**
23  * Receives events and sends them to subscribed clients.
24  * This is the backbone of the runtime control messaging
25  * system connecting the various modules.
26  *
27  * We identify an event by a plain vanilla integer constant. Java texbook approach
28  * would dictate using an enumerator or a dedicated class hierarchy for events.
29  * We deviate from the textbook approach, because both approaches involve creating
30  * objects. That's fine for the occasional event. However, for a constant stream
31  * of events (such as generated by the {@link TTimer} class) this results in ongoing
32  * creation of throw-away objects many times per second. This smells like a silly design.
33  * Therefore we resort to the out-of-date technique of using integer constants. This
34  * is much more performant.
35  *
36  * @author Peter Hoppe
37  * @see IEvented
38  */
39 public class TBroker

124  /**
125  * Internal broker worker method.
126  *
127  * @param e The event being brokered.
128  * @param idSource ID of the module that has fired the event.
129  * @param argsType Which type is the in-band parameter (none, int, long, String)?
130  * @param arg_0_0 If in-band parameter is an int: Its value.
131  * @param arg_0_1 If in-band parameter is a long: Its value.
132  * @param arg_1_0 If in-band parameter is a String: Its value.
133  */
134  private void _Broker (int e, String idSource, int argsType, int arg_0_0, long arg_0_1, String arg_1_0)
135  {
136  TBrokerSubscription subscr;
137  TBrokerSubscriptionList subscrList;
138  int i;
139  int n;
140  IEvented receiver;
141 
142  subscrList = fSubscriptions.Get (idSource);
143  n = subscrList.GetNumElements ();
144  if (n >= 1)
145  {
146  for (i = 0; i < n; i++)
147  {
148  subscr = subscrList.Get (i);
149  receiver = (IEvented) subscr.GetSubscriber ();
150  if (argsType == gkArgsNone) /* [100] */
151  {
152  receiver.OnEvent (e);
153  }
154  else if (argsType == gkArgsOneInt)
155  {
156  receiver.OnEvent (e, arg_0_0);
157  }
158  else if (argsType == gkArgsOneLong)
159  {
160  receiver.OnEvent (e, arg_0_1);
161  }
162  else if (argsType == gkArgsOneString)
163  {
164  receiver.OnEvent (e, arg_1_0);
165  }
166  }
167  }
168  }

170  /**
171  * Subscribes an {@link IEvented} object (subscriber) to receive events
172  * from another {@link VBrowseable} (subscribed). The subscriber will
173  * receive all events emitted by the subscribed and needs to implement
174  * its own event filtering to listen to the events it's interested in.
175  *
176  * @param subscribed Subscribed.
177  * @param subscriber Subscriber.
178  */
179  public void Subscribe (VBrowseable subscribed, IEvented subscriber)
180  {
181  TBrokerSubscriptionList subscrList;
182  TBrokerSubscription subscr;
183  String key;
184  boolean hasKey;
185 
186  key = subscribed.GetID ();
187  hasKey = fSubscriptions.HasElement (key);
188  if (hasKey)
189  {
190  subscrList = fSubscriptions.Get (key);
191  }
192  else
193  {
194  subscrList = new TBrokerSubscriptionList (subscribed);
195  fSubscriptions.Set (key, subscrList);
196  }
197 
198  subscr = new TBrokerSubscription (subscribed, subscriber);
199  subscrList.Add (subscr);
200  }
201 }
