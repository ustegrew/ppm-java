72  /**
73  * Sets up a session with the given session properties.
74  *
75  * @param sessionProps The session properties to use.
76  */
77  public void ExecuteWith (TSessionProperties sessionProps)
78  {
79  String helpText;
80 
81  fOptConsoleWidth = sessionProps.GetOptConsoleWidth ();
82  fOptLogFilePath = sessionProps.GetOptLogFilePath ();
83  fOptPrintHelp = sessionProps.IsOptPrintHelp ();
84  fOptShowDebugWindow = sessionProps.IsOptShowDebugWindow ();
85  fOptUIType = sessionProps.GetOptUIType ();
86  fOptUseDeprecatedPPMProcessor = sessionProps.IsOptUseDeprecatedPPMProcessor ();
87 
88  if (! fOptPrintHelp)
89  {
90  TLogger.CreateInstance (fOptLogFilePath);
91  if (fOptUseDeprecatedPPMProcessor)
92  {
93  _Setup_UseDeprecatedPPMProcessor ();
94  }
95  else
96  {
97  _Setup ();
98  }
99  }
100  else
101  {
102  helpText = sessionProps.GetHelpText ();
103  System.out.println (helpText);
104  }
105  }
