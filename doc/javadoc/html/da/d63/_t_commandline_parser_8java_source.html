62  /**
63  * Parses the given commandline arguments. Show help text if
64  * there's a problem with the arguments.
65  *
66  * @param args Given commandline arguments.
67  * @return The session properties, or <code>null</code> if there's a problem.
68  */
69  public TSessionProperties Parse (String[] args)
70  {
71  boolean isOK;
72  CommandLineParser clp;
73  CommandLine cl;
74  TSessionProperties ret;
75 
76  /* Parse commandline */
77  isOK = true;
78  cl = null;
79  ret = new TSessionProperties ();
80  ret.SetHelpText (fHelpText);
81  try
82  {
83  clp = new DefaultParser ();
84  cl = clp.parse (fCommandlineOptions, args);
85  }
86  catch (ParseException e)
87  {
88  isOK = false;
89  ret.SetErrorMessage (e.getMessage ());
90  }
91 
92  /* Parsing complete. Set program properties. */
93  if (isOK)
94  {
95  ret.Set (cl);
96  isOK = ret.IsValid (); /* [100] */
97  }
98 
99  return ret;
100  }

164  80, /* number of characters to be displayed on each line */
165  gkCommandlineSyntax, /* syntax for this application */
166  "", /* banner to display at the beginning of the help */
167  fCommandlineOptions, /* Options instance */
168  4, /* number of characters of padding to be prefixed to each line */
169  4, /* the number of characters of padding to be prefixed to each description line */
170  "", /* the banner to display at the end of the help */
171  true /* whether to print an automatically generated usage statement */
172  );
