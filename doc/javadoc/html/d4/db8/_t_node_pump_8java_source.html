32 /**
33  * A data pump. Fetches the audio data from the audio driver,
34  * repackages it in smaller chunks if needed and passes those
35  * on to further stages. Also, initiates cycle time correction
36  * to the GUI timer if we have data loss.<br/>
37  * The data pump is designed to work together with an audio driver;
38  * it acts as a priority inversion barrier between audio driver
39  * and display engine. This enables the audio driver to work
40  * undisturbed, no matter what happens on the display driver side.
41  *
42  * @author Peter Hoppe
43  */

262  /**
263  * Recomputes some runtime statistics for the next
264  * cycle (e.g. how many samples we should pass on).
265  */
266  private void _RecomputeRuntimeStats ()
267  {
268  long tNow;
269  long tDelta;
270  double xSampleRate;
271  double nSamplesPerCycle;
272 
273  tNow = System.currentTimeMillis ();
274  tDelta = tNow - fTLast;
275  if (fTCycleDesired <= -1)
276  {
277  fTCycleDesired = tDelta;
278  }
279  xSampleRate = fAudioDriver.GetSampleRate ();
280  nSamplesPerCycle = tDelta * xSampleRate / 1000; /* [140] */
281 
282  fTDelta = tDelta;
283  fTLast = tNow;
284  fSampleRate = (int) xSampleRate;
285  fNSamplesPerCycle = (int) (nSamplesPerCycle);
286 
287  fStats.SetCycleTime (fTDelta);
288  fStats.SetNumSamplesPerCycle (fNSamplesPerCycle);
289  fStats.SetSampleRate (fSampleRate);

293  /**
294  * Compensates for data loss (i.e. under/overruns) on the
295  * attached input buffer by issuing a request to the GUI timer
296  * to set the timer interval to a new value.
297  * If we have underruns this means that we aren't getting
298  * enough data which means the cycle interval is too short
299  * and needs to be longer. If we have overruns it means
300  * that the audio driver is delivering more data than we
301  * can process which means the cycle interval is too long.<br/>
302  * Correction follows a simple linear strategy - for each
303  * cycle, if we detected data loss we increase or decrease
304  * the desired cycle time by one millisecond and notify
305  * the GUI timer that we request a change of cycle time.
306  */
307  private void _Resolve_DataLoss ()
308  {
309  String id;
310  TAtomicBuffer_Stats st;
311  TAtomicBuffer_DumpRecord stRec;
312 
313  id = GetID ();
314  st = fEndptIn.StatsGet ();
315  stRec = st.GetRecord ();
316 
317  if (stRec.fDiffOverUnderruns > 0)
318  { /* Overruns - meaning the JackD audio driver pushes more
319  data than we handle at the moment => We must decrement
320  GUI timer cycle interval. We use a simple linear decrement. */
321  if (fTCycleDesired > TTimer.gkLoopIntervalMin)
322  {
323  fTCycleDesired--;
324  TController.PostEvent (gkEventTimerAdjustInterval, fTCycleDesired, id);
325  }
326  st.Clear (); /* [130 */
327  }
328  else if (stRec.fDiffOverUnderruns < 0)
329  { /* Underruns - meaning the GUI update cycle tries to draw too
330  * much data => We must increment GUI timer cycle interval.
331  * We use a simple linear increment. */
332  fTCycleDesired++;
333  TController.PostEvent (gkEventTimerAdjustInterval, fTCycleDesired, id);
334  st.Clear (); /* [130] */
335  }
336  }

338  /**
339  * Retrieves next chunk of data. Since our cycle frequency is different
340  * than the frequency in which the JackD driver delivers data we
341  * need to chop the incoming packets up to process the same amount of
342  * samples per time as what's delivered by the audio driver.
343  */
344  private void _SendNextPacket ()
345  {
346  FloatBuffer newPacket;
347  int i;
348  int nSamples;
349  float s;
350  FloatBuffer outPacket;
351 
352  newPacket = fEndptIn.FetchPacket ();
353  if (newPacket != null)
354  {
355  /* If we have new data we discard any old data */
356  fPacketLast = newPacket; /* Discard previous packet. */
357  fPacketLast.rewind (); /* Ensure read pointer and markers are at zero position. */
358  nSamples = fPacketLast.limit (); /* How many samples are we talking about? */
359  if (nSamples > fNSamplesPerCycle) /* Too many samples? */
360  { /* For now, only read what we need. Postpone rest for next cycle(s) */
361  nSamples = (int) fNSamplesPerCycle;
362  }
363  }
364  else if (fPacketLast != null)
365  {
366  /* No fresh data available, but the previous packet may still have unresolved data */
367  nSamples = fPacketLast.remaining ();
368  if (nSamples > fNSamplesPerCycle)
369  { /* We have too much data to crunch. Only use as much as needed. */
370  nSamples = (int) fNSamplesPerCycle;
371  }
372  else if (nSamples <= 0)
373  { /* No data left to read. Delete packet. */
374  fPacketLast = null;
375  }
376  /* else-case: All the possible values in [1, fNSamplesPerCycle] */
377 
378  /* At this point, nSamples will be between 0 and fNSamplesPerCycle. */
379  }
380  else
381  { /* No data available */
382  nSamples = 0;
383  }
384 
385  if (nSamples >= 1)
386  {
387  /* We've got samples to send. Assemble an output packet (= new object). [120] */
388  outPacket = FloatBuffer.allocate (nSamples);
389  for (i = 1; i <= nSamples; i++)
390  {
391  s = fPacketLast.get ();
392  outPacket.put (s);
393  }
394  }
395  else
396  {
397  /* We've got no samples to send. Send a null packet. */
398  outPacket = null;
399  }
400 
401  /* Push sample data to the next module. */
402  fEndptOut.PushPacket (outPacket);
403  }
404 }

406 /*
407 [100] For time measurement. We will use the first cycle after starting
408  to just set an initial time point and otherwise do no data processing.
409  Otherwise, the first cycle could provide non-sensical values, maybe
410  even leading to an exception.
411 [110] For the next packet we read as many samples as we set in the
412  previous step (fNSamplesPerCycle).
413 [120] Unfortunately, this means we have to copy data again which I wanted
414  to avoid. One idea was to postpone the copying to by instructing the
415  atomic buffer to omit copying and to do the copying when we assemble
416  the outgoing packets here. However, this doesn't work. By the time
417  we copy the data here, the atomic buffer has released it's critical
418  sections and the producer might rewrite the buffer which will give us
419  faulty data here. Unfortunately there isn't an easy way around this.
420  It was this place which inspired the TAtomicBuffer kNoCopy policy, and the
421  only place where this policy would make some sense. We leave the policy
422  in place, just in case we find a way to use it without compromising
423  thread safety. For future developments. For now, the extra copy doesn't
424  seem to have a visible impact on the GUI behaviour.
425 [130] Otherwise stRec.fDiffOverUnderruns won't converge to zero without more over/underruns
426 [140] Imbalance problem: Number of samples is based on GUI cycle time and sampling frequency.
427  We will get a situation where during one GUI cycle we process a majority of the last
428  Jack frame; then in the next GUI cycle we compute process that frame's leftovers
429  (as Jack hasn't delivered a new frame yet). The first GUI cycle will have a lot of
430  samples, the next GUI cycle will have a few remaining samples.
431  Mitigation: We need to compute the GUI cycle number of samples in such a way that this
432  imbalance is minimized -> the calculation must include the Jack frame size, dividing
433  the Jack frame evenly among the successive GUI cycles as known at the time of calculation.
434  This calculation must be renewed each GUI cycle, i.e. happen inside _RecomputeRuntimeStats().
435  This means, we need to modify the calculation at the source line with footnote [140]
436  to include Jack frame size and approx. evenly distribute sample chunk size per GUI cycle.
437  It'll never be totally even, as the number of samples can only be an integer, and
438  "really even" would mean a real number, i.e. in the mathematical set of real numbers.
439  You simply can't have a "real number" of samples, only a "Natural{0, 1, 2, ...} number"
440  of samples.
441  */
