22 /**
23  * Debug update worker thread. Updates the runtime stats window. Polls
24  * the statistics in a regular cycle ({@link #gkInterval}) and writes
25  * them to the associated debug {@link TWndDebug UI}.
26  *
27  * @author peter

29 public class TTimerDebugUpdate extends Thread
30 {
31  /**
32  * Flag value: Thread running.
33  */
34  private static final int gkFlagRun = 1;

36  /**
37  * Flag value: Thread stopped.
38  */
39  private static final int gkFlagStop = 0;
40 
41  /**
42  * Cycle's interval, in ms.
43  */
44  private static final int gkInterval = 10;
45 
46  /**
47  * Flag. Set to {@link #gkFlagStop} when we want to terminate the thread.
48  */
49  private AtomicInteger fDoRun;
50 
51  /**
52  * cTor.

54  public TTimerDebugUpdate ()

57  }
58 
59  /**
60  * Terminates the worker thread.
61  */
62  public void Stop ()

65  }
66 
67  /* (non-Javadoc)
68  * @see java.lang.Thread#start()
69  */
70  @Override
71  public void run ()
72  {
73  String stats;
74  int doRun;
75 
76  TWndDebug.Show ();
77 
78  doRun = gkFlagRun;
79  fDoRun.getAndSet (gkFlagRun);
80  while (doRun == gkFlagRun)
81  {
82  stats = TController.StatGetDumpStr ();
83  TWndDebug.SetText (stats);
84  try {Thread.sleep (gkInterval);} catch (InterruptedException e) {}
85  doRun = fDoRun.addAndGet (0);
86  }
87  }
